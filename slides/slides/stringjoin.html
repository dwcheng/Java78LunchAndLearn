String Join 

String[] names = {"John", "Josh", "Bob", "Alex"};

StringBuilder stringBuilder = new StringBuilder(names.length);

for (int i = 0; i < names.length; i++) {
    stringBuilder.append(names[i]);
    if (i < names.length - 1) {
        stringBuilder.append(", ");
    }
}

return stringBuilder.toString();
