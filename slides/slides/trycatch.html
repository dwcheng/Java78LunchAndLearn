Try Catch With Resources 

InputStream inputStream = null;
try {
    inputStream = new FileInputStream(new File("foo.txt"));
    // Do stuff
} catch (FileNotFoundException e) {
    // Handle FileNotFoundException
} finally {
    try {
        inputStream.close();
    } catch (IOException e) {
        // Handle IOException
    }
}
Try Catch With Resources 

try (InputStream inputStream = new FileInputStream(new File("foo.txt"))) {
    // Do stuff
} catch (IOException e) {
    // Handle exception
}
